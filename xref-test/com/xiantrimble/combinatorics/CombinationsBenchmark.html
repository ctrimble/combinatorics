16  package com.xiantrimble.combinatorics;

18  import java.util.Arrays;
19  import java.util.NoSuchElementException;
20  
21  public class CombinationsBenchmark<T>
22    extends AbstractCombinatoric<T>
23  {
24    public CombinationsBenchmark(int k, T[] domain, CombMathUtils mathUtil) {
25      super(k, domain, mathUtil);
26    } 
27  
28    @Override
29    public CombinatoricIterator<T> iterator() {
30      return new CombinationIterator(0);
31    }
32  
33    @Override
34    protected long computeSize(int k, GroupedDomain<T> domain) {
35      return mathUtils.c(k, domain.toMultiplicity());
36    }
37    
38    protected class CombinationIterator
39      extends AbstractCombinatoric<T>.AbstractCombinatoricIterator
40    {
41      protected T[] next;
42      protected T[] previous;
43      protected int[] domainMultiplicity;
44      protected DomainPointer[] indices;
45      
46      protected CombinationIterator(long nextIndex) {
47        super(nextIndex);
48        next = newComponentArray(k);
49        previous = newComponentArray(k);
50        domainMultiplicity = domain.toMultiplicity();
51        indices = new DomainPointer[domainMultiplicity.length];
52        indices[indices.length-1] = new DomainPointer();
53        for( int i = domainMultiplicity.length-1; i > 0; i-- ) {
54          indices[i-1] = new DomainPointer();
55          indices[i-1].toRight = domainMultiplicity[i] + indices[i].toRight;
56        }      
57      }
58  
59      @Override
60      public T[] next() {
61        if( nextIndex >= size ) throw new NoSuchElementException(); // we may just want to do this in the next method.
62        
63        // reset the next array if needed.
64        if( nextIndex == 0 ) {
65          for(int i = 0, used = 0; i < indices.length && used < k; used += domainMultiplicity[i++]) {
66            indices[i].index = used;
67            indices[i].count = Math.min(k-used, domainMultiplicity[i]);
68            for( int j = 0; j < indices[i].count; j++ ) {
69              next[indices[i].index+j] = domainValues[i][j];
70            }
71          }
72        }
73        
74        // set both values to the the next value.
75        for( int i = 0; i < next.length; i++ ) previous[i] = next[i];
76        nextIndex++;
77        
78        if( nextIndex != size ) {
79        
80        // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION: (3,2,0,1,1)
81        // Position  | 0 | 1 | 2 | 3 | 4
82        // Index     | 0 | 3 | 5 | 5 | 6
83        // Count     | 3 | 2 | 0 | 1 | 1
84        // ToRight   | 9 | 6 | 5 | 2 | 0
85  
86        // advance the indices.
87        int cur = domainMultiplicity.length - 1;
88        int remaining = 0;
89        
90        // move cur backwards to find the next item to update.
91        for( ; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining+1); remaining += indices[cur--].count ); // back cur up to a position to increment.
92        
93        // decrement the items at cur.
94        indices[cur].count--;
95        remaining++;
96        
97        // move forward and update indices and next.
98        for(cur++; cur < indices.length; cur++) {
99          indices[cur].count = Math.min(remaining, domainMultiplicity[cur]);
100         indices[cur].index = indices[cur-1].index+indices[cur-1].count;
101         remaining -= indices[cur].count;
102         for( int i = 0; i < indices[cur].count; i++ ) {
103           next[indices[cur].index+i] = domainValues[cur][i];
104         }
105       }
106       }
107       // return previous, since we advanced past the next position.
108       return Arrays.copyOf(previous, previous.length);
109     }
110 
111     @Override
112     public T[] previous() {
113       if( nextIndex <= 0 ) throw new NoSuchElementException(); // we may just want to do this in the next method.
114       
115       // set both values to the the previous value.
116       for( int i = 0; i < previous.length; i++ ) next[i] = previous[i];
117       nextIndex--;
118       
119       if( nextIndex > 0 ) {
120       
121       // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION: (3,2,0,1,1)
122       // Position  | 0 | 1 | 2 | 3 | 4
123       // Index     | 0 | 3 | 5 | 5 | 6
124       // Count     | 3 | 2 | 0 | 1 | 1
125       // ToRight   | 9 | 6 | 5 | 2 | 0
126 
127       // advance the indices.
128       int cur = domainMultiplicity.length - 1;
129       int remaining = 0;
130 
131       // move cur backwards to find an item to increment.
132       for( ; cur > 0 && (remaining == 0 || indices[cur].count == domainMultiplicity[cur]); remaining += indices[cur--].count );
133 
134       // decrement the items at cur.
135       indices[cur].count++;
136       previous[indices[cur].index+indices[cur].count-1] = domainValues[cur][indices[cur].count-1];
137       remaining--;
138        
139       for(cur++; cur < indices.length; cur++) {
140         indices[cur].count = Math.max(remaining-indices[cur].toRight, 0);
141         indices[cur].index = indices[cur-1].index+indices[cur-1].count;
142         remaining -= indices[cur].count;
143         for( int i = 0; i < indices[cur].count; i++ ) {
144           previous[indices[cur].index+i] = domainValues[cur][i];
145         }
146       }
147       }
148       
149       // return next, since we advanced past the previous position.
150       return next;
151     }
152   }

154   private static class DomainPointer
155   {
156     public int index = 0;
157     public int count = 0;
158     public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
159     
160     public String toString()
161     {
162       return "{Index:"+index+",Count:"+count+",ToRight:"+toRight+"}";
163     }
164   }
