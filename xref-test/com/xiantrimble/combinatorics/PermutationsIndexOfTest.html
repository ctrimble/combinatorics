16  package com.xiantrimble.combinatorics;
17  
18  import static org.junit.Assert.*;
19  
20  import java.util.Arrays;
21  import java.util.Collection;
22  
23  import org.junit.Test;
24  import org.junit.runner.RunWith;
25  import org.junit.runners.Parameterized;
26  import org.junit.runners.Parameterized.Parameters;
27  
28  import static com.xiantrimble.combinatorics.PermutationsIndexOfTest.Element.*;
29  
30  @RunWith(Parameterized.class)
31  public class PermutationsIndexOfTest {
32    public static enum Element {
33      ONE,
34      TWO,
35      THREE,
36      FOUR,
37      FIVE
38    }
39    private static CombinatoricFactory factory = new CombinatoricFactoryImpl();
40  
41    private Combinatoric<Element> combinatoric;
42  
43    public PermutationsIndexOfTest( Combinatoric<Element> combinatoric ) {
44      this.combinatoric = combinatoric;
45    }
46    
47    @Test
48    public void iterationMatchesIndexOf() {
49      int index = 0;
50      for( Element[] element : combinatoric) {
51        long indexOf = combinatoric.longIndexOf(element);
52        //System.out.println("index "+index+" "+Arrays.toString(element));
53        try {
54          assertEquals(index, indexOf);
55        }
56        catch( AssertionError ae ) {
57          // handle for debugger.
58          combinatoric.longIndexOf(element);
59          throw ae;
60        }
61        index++;
62      }
63    }
64  
65    @Parameters
66    public static Collection<Object[]> parameters() {
67      return Arrays.asList(new Object[][] {
68          { permutaitons( 1, ONE ) },
69          { permutaitons( 2, ONE, TWO ) },
70          { permutaitons( 3, ONE, TWO, TWO, THREE, THREE, THREE, FOUR, FOUR, FOUR ) },
71          { permutaitons( 10, ONE, ONE, ONE, TWO, TWO, TWO, THREE, THREE, THREE, FOUR, FOUR, FOUR ) }
72      });
73    }
74    
75    public static <E> Combinatoric<E> combinations(int k, E... domain) {
76      return factory.createCombinations(k, domain);
77    }
78    
79    public static <E> Combinatoric<E> permutaitons(int k, E... domain) {
80      //return factory.createPermutations(k, domain);
81      return new IndexBasedPermutations(k, domain, factory.getMathUtils());
82    }  
83  }
