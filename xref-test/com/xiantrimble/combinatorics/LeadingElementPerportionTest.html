33  /**
34   * This test verifies that for permutations of length k, containing two unique elements and k
35   * total elements, the percentage of the time that any element is at any position is equal
36   * to the percentage of the domain it represents.
37   * 
38   * This fact has implications for computing the size of a permutation, when you know
39   * the size of some permutation near it.
40   * 
41   * Let kP be a permutation with a known k (kK), known multiplicity (kM[]), and known size (kS).
42   * Let p be a permutation that varies from kP by type removal of some element from kM[i], with a known k (k=kK-1),
43   * a known multiplicity (m[]), and an undetermined size (s).
44   *
45   * Then, this will be true:
46   * s = ( kM[i] / total(kM) ) * kS
47   * 
48   * 
49   * @author Christian Trimble
50   *
51   */
52  @RunWith(Parameterized.class)
53  public class LeadingElementPerportionTest {
54    public static enum Element {
55      ONE,
56      TWO,
57      THREE
58    }
59    private static CombinatoricFactory factory = new CombinatoricFactoryImpl();
60    private Combinatoric<Element> permutations;
61    
62    public LeadingElementPerportionTest( Combinatoric<Element> permutations ) {
63      this.permutations = permutations;
64    }
65  
66    @Test
67    public void twoElementsPerportional() {
68      int[] leadingOnes = new int[permutations.getK()];
69      int[] leadingTwos = new int[permutations.getK()];
70      for( Element[] element : permutations ) {
71        for( int i = 0; i < element.length; i++ ) {
72          leadingOnes[i] += element[i]==ONE?1:0;
73          leadingTwos[i] += element[i]==TWO?1:0;
74        }
75      }
76      
77      // Turn the leading arrays into proper fractions.
78      int[] leadingGcd = new int[permutations.getK()];
79      for( int i = 0; i < permutations.getK(); i++ ) {
80        leadingGcd[i] = MathUtils.gcd(leadingOnes[i], leadingTwos[i]);
81        leadingOnes[i] /= leadingGcd[i];
82        leadingTwos[i] /= leadingGcd[i];
83      }
84      
85      int totalOnes = permutations.getDomain().get(0).size();
86      int totalTwos = permutations.getDomain().get(1).size();
87      
88      // Turn the total into a proper fraction.
89      int totalGcd = MathUtils.gcd(totalOnes, totalTwos);
90      totalOnes /= totalGcd;
91      totalTwos /= totalGcd;
92      
93      for( int i = 0; i < permutations.getK(); i++ ) {
94        assertEquals("The wrong perportion of ones were found at index "+i, totalOnes, leadingOnes[i]);
95        assertEquals("The wrong perportion of twos were found at index "+i, totalTwos, leadingTwos[i]);
96      }
97    }

100   /**
101    * if we make kM[i]/kS a proper fraction, can we always apply the denominator first to kS?
102    * This test demonstrates that this is safe.  Applying this first will allow us to
103    * approach Long.MAX_LONG sizes safely.
104    */
105   @Test
106   public void alwaysDivisible() {
107     long size = permutations.size();
108     int kTotalM = permutations.getDomain().totalSize();
109     for( int m : permutations.getDomain().toMultiplicity() ) {
110       long gcd = MathUtils.gcd(kTotalM, m);
111       long den = kTotalM / gcd;
112       assertEquals("Greatest common divisor is not denominator", den, MathUtils.gcd(size, den));
113     }
114   }
115 
116   @Test
117   public void relativeComputation() {
118    long kS = permutations.size();
119    int kTotalM = permutations.getDomain().totalSize();
120    int[] kM = permutations.getDomain().toMultiplicity();
121    for( int i = 0; i < kM.length; i++ ) {
122      long gcd = MathUtils.gcd(kTotalM, kM[i]);
123      long num = kM[i] / gcd;
124      long den = kTotalM / gcd;
125      long s = (kS / den) * num;
126      
127      // compute the expected result.
128      int[] m = Arrays.copyOf(kM, kM.length);
129      m[i]--;
130      long expected = factory.getMathUtils().p(permutations.getK()-1, m);
131      
132      // verify that the computed size is the same as the expected size.
133      assertEquals(expected, s);
134    }
135   }
