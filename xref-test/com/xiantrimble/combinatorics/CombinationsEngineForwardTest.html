18  import java.util.ArrayList;
19  import java.util.Arrays;
20  import java.util.List;
21  
22  public class CombinationsEngineForwardTest extends AbstractCombinationIteratorTest {
23  
24    @Override
25    public List<List<Integer>> createCombinations(List<Integer> elements, int k) {
26      CombinatoricFactory factory = new CombinatoricFactoryImpl();
27      CombinatoricEngine<Integer> combinationEngine = factory.createCombinationsEngine(k,  (Integer[])elements.toArray(new Integer[elements.size()]));
28      final List<List<Integer>> result = new ArrayList<List<Integer>>();
29      final Integer[] state = new Integer[k];
30      combinationEngine.setHandler(new AbstractCombinatoricHandler<Integer>() {
31        @Override
32        public void evaluate() {
33          result.add(Arrays.asList(Arrays.copyOf(state, state.length))); 
34        }
35  
36        @Override
37        public void init(Integer[] newState ) {
38          for( int i = 0; i < newState.length; i++ ) {
39            state[i] = newState[i];
40          }
41        }
42  
43        @Override
44        public void swap(Integer newA, int ai, Integer newB, int bi) {
45          state[ai] = newA;
46          state[bi] = newB;
47        }
48  
49        @Override
50        public void replace(Integer newValue, Integer oldValue, int i) {
51          state[i] = newValue;
52        }
53      });
54      combinationEngine.execute();
55  
56      return result;
57    }
