16  package com.github.ctrimble.combinatorics;
17  
18  import java.util.List;
19  
20  /**
21   * A representation of a set of elements where the like elements have been grouped.  Usually, a 
22   * multiset would just contain information about the quantities of the distinct types in a collection,
23   * but this implementation also holds the actual elements.
24   * 
25   * @author Christian Trimble
26   *
27   * @param <E> the element type of the collection or array that this multiset represents.
28   */
29  public interface Multiset<E> extends List<List<E>>
30  {
31    /**
32     * Returns the total number of elements in this multiset.
33     * 
34     * @return the total number of elements in this multiset.
35     */
36    public int getRank();
37    
38    /**
39     * Returns an array of the element ranks for this multiset.
40     * 
41     * @return an array of the element ranks for this multiset.
42     */
43    public int[] toRankArray();
44    
45    /**
46     * Returns a two dimensional array representation of this multiset.
47     * 
48     * @return a two dimensional array representation of this multiset.
49     */
50    public E[][] toValueArray();
51  }
