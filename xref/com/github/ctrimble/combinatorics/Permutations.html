21  /**
22   * A collection class that contains the permutations of a set of elements.  This implementation accounts for duplicate elements in the domain.
23   * 
24   * @author Christian Trimble
25   *
26   * @param <T> The type of element being permuted.
27   */
28  public class Permutations<T>
29    extends AbstractCombinatoric<T>
30  {
31  
32    /**
33     * Creates a new Permutations object over the domain.
34     * 
35     * @param rank the length of the permutations.
36     * @param domain the elements that make up the permutations.
37     * @param mathUtils an instance of the math utilities.
38     */
39    protected Permutations(int rank, T[] domain, CombMathUtils mathUtils) {
40      super(rank, domain, mathUtils);
41    }
42   
43    @Override
44    public CombinatoricIterator<T> iterator() {
45      return new PermutationIterator(0);
46    }

48    /**
49     * Computes the number of permutations for the specified rank and domain.
50     * 
51     * @param rank the rank of the permutation.
52     * @param domain the multiset containing the elements to be permuted.
53     */
54    @Override
55    protected long computeSize(int rank, Multiset<T> domain) {
56      return mathUtils.p(rank, domain.toRankArray());
57    }

59    /**
60     * An iterator for this permutation.
61     * 
62     * @author Christian Trimble
63     */
64    public class PermutationIterator
65      extends AbstractCombinatoricIterator
66    {
67      TypePermutationState[] state;
68      int[] domainRanks;
69      T[] next;
70      T[] previous;
71  
72      protected PermutationIterator(int nextIndex) {
73        super(nextIndex);
74      }
75  
76      /**
77       * Returns the next permutation, if one exists.
78       * 
79       * @throws NoSuchElementException if there are no more permutations in this collection.
80       */
81      @Override
82      public T[] next() {
83        if( nextIndex >= size ) throw new NoSuchElementException();
84        // if the index is 0, build the next array and the iteration state.
85        if( nextIndex == 0 ) {
86          next = newComponentArray(rank);
87          domainRanks = domain.toRankArray();
88          state = new TypePermutationState[domainRanks.length];
89          int ni = 0;
90          for( int ri = 0; ri < domainRanks.length; ri++) {
91            state[ri] = new TypePermutationState();
92            state[ri].count = Math.min(domainRanks[ri], rank-ni);
93            state[ri].entryState = new EntryPermutationState[domainRanks[ri]];
94            for( int j = 0; j < state[ri].entryState.length; j++ ) {
95              state[ri].entryState[j] = new EntryPermutationState(j);
96              if( j < state[ri].count ) {
97                next[ni++] = domain.get(ri).get(j);         
98              }
99            }
100         }
101         for( int i = state.length - 2; i >= 0; i-- ) {
102           state[i].toRight = state[i].entryState.length + state[i+1].toRight;
103         }
104       }
105       nextIndex++;
106 
107       // clone the next array into the previous array.
108       previous = Arrays.copyOf(next, next.length);
109       
110       if( nextIndex == size ) {
111         next = null;
112         return previous;
113       }
114       
115       int windowStart = 0;
116       int windowEnd = next.length;
117       int swapSource = 0;
118       int swapTarget = 0;
119       TYPE: for( int i = 0; i < state.length; i++ ) {
120         switch( state[i].direction ) {
121           case DOWN:
122             // scan the entries from back to front, looking for the first item to move.
123             ENTRY: for( int j = state[i].count-1; j >= 0; j-- ) {
124               switch (state[i].entryState[j].direction) {
125                 case DOWN:
126                   if( state[i].entryState[j].index < (windowEnd - 1) - ((state[i].count-1)-j) ) {
127                     // track the source index for the swap.
128                     swapSource = windowStart + state[i].entryState[j].index;
129                        
130                     // increment the index and deal with moving over like elements
131                     state[i].entryState[j].index++;
132                     for(;j < state[i].count-1 && state[i].entryState[j].index == state[i].entryState[j+1].index; j++ ) {
133                       state[i].entryState[j+1].index++;
134                       state[i].entryState[j+1].direction = Direction.DOWN;
135                     }
136 
137                     // track the target index for the swap.
138                     swapTarget = windowStart + state[i].entryState[j].index;
139                     break TYPE;
140                   }
141                   state[i].entryState[j].direction = Direction.UP;
142                   break;
143                 case UP:
144                     int startIndex = j;
145                     for(; state[i].entryState[j].index == state[i].entryState[j-1].index+1; j-- ) {
146                       // if this entry is moving down, then switch case statements.
147                       if( state[i].entryState[j-1].direction == Direction.DOWN ) {
148                         continue ENTRY;
149                       }
150                     }
151                     swapSource = windowStart + state[i].entryState[startIndex].index;
152                     state[i].entryState[j].direction = Direction.UP;
153                     state[i].entryState[j].index--;
154                     swapTarget = windowStart + state[i].entryState[j].index;
155                     for(j++; j <= startIndex; j++) {
156                       state[i].entryState[j].direction = Direction.DOWN;
157                       state[i].entryState[j].index--;
158                     }
159                     break TYPE;
160                 }
161               }
162             // none of the entries can move down, switch directions.
163             state[i].direction = Direction.UP;
164             windowEnd -= state[i].count;
165             if( i < state.length - 1 ) continue TYPE;
166             break;
167           case UP:
168             // scan the entries from front to back, looking for the first item to move.
169             ENTRY:for( int j = 0; j < state[i].count; j++ ) {
170               switch (state[i].entryState[j].direction) {
171                 case UP:
172                   if( state[i].entryState[j].index > j ) {
173                     // track the source index of the swap.
174                     swapSource = state[i].entryState[j].index+ windowStart;
175                     
176                     // decrement the index and deal with moving over like elements.
177                     state[i].entryState[j].index--;
178                     for(; j > 0 && state[i].entryState[j].index == state[i].entryState[j-1].index; j--) {
179                       state[i].entryState[j-1].index--;
180                       state[i].entryState[j-1].direction = Direction.UP;
181                     }
182                     
183                     // track the target index for the swap.
184                     swapTarget = state[i].entryState[j].index + windowStart;
185                     break TYPE;
186                   }
187                   state[i].entryState[j].direction = Direction.DOWN;
188                   break;
189                 case DOWN:
190                   int startIndex = j;
191                   for(; state[i].entryState[j].index == state[i].entryState[j+1].index-1; j++ ) {
192                     // if this entry is moving down, then switch case statements.
193                     if( state[i].entryState[j+1].direction == Direction.UP ) {
194                       continue ENTRY;
195                     }
196                   }
197                   swapSource = windowStart + state[i].entryState[startIndex].index;
198                   state[i].entryState[j].direction = Direction.DOWN;
199                   state[i].entryState[j].index++;
200                   swapTarget = windowStart + state[i].entryState[j].index;
201                   for(j--; j >= startIndex; j--) {
202                     state[i].entryState[j].direction = Direction.UP;
203                     state[i].entryState[j].index++;
204                   }
205                   break TYPE;
206               }
207             }
208             state[i].direction = Direction.DOWN;
209             windowStart += state[i].count;
210             if( i < state.length - 1 ) continue TYPE;
211             break;
212         }
213         
214         // end of permutations for this combination reached.  Advance to the next combination.
215         int cur = state.length - 1;
216         int remaining = 0;
217         
218         // move cur backwards to find the next item to update.
219         for( ; cur > 0 && (state[cur].count == 0 || state[cur].toRight < remaining+1); remaining += state[cur--].count ); // back cur up to a position to increment.
220         
221         // decrement the items at cur.
222         state[cur].count--;
223         remaining++;
224         
225         // move forward and update all of the counts.
226         for(cur++; cur < state.length; cur++) {
227           state[cur].count = Math.min(remaining, domainRanks[cur]);
228           remaining -= state[cur].count;
229         }
230         
231         // for now, totally reset next.  Making this an incremental update will help when the rank is much smaller than the number of entries.
232         for( int ri = 0, ni = 0;ri < domainRanks.length; ri++) {
233           state[ri].direction = Direction.DOWN;
234           for( int k = 0; k < state[ri].entryState.length; k++ ) {
235             state[ri].entryState[k].index = k;
236             state[ri].entryState[k].direction = Direction.DOWN;
237             if( k < state[ri].count ) {
238               next[ni++] = domain.get(ri).get(k);         
239             }
240           }
241         }
242       }
243       
244       // swap the indices.
245       T target = next[swapTarget];
246       next[swapTarget] = next[swapSource];
247       next[swapSource] = target;
248       
249       //System.out.println(Arrays.toString(previous));
250       
251       return previous;
252     }

266   private static class TypePermutationState
267   {
268     public Direction direction = Direction.DOWN;
269     public EntryPermutationState[] entryState;
270     public int count = 0; // the number of entries being used.
271     public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
272     public String toString() {
273       return "{direction:"+direction+", entries:"+Arrays.toString(entryState)+"}";
274     }
275   }

277   private static class EntryPermutationState
278   {
279     public EntryPermutationState(int index) {
280       this.index = index;
281     }
282     public Direction direction = Direction.DOWN;
283     public int index = 0;
284     public String toString() {
285       return "{direction:"+direction+", index:"+index+"}";
286     }
