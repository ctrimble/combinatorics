4     protected T[] last;

8     protected CombinationsEngine(int rank, T[] domain, CombMathUtils mathUtils) {
9       super(rank, domain, mathUtils);
10      last = Utils.newArray(componentType, rank);
11      domainRanks = this.domain.toRankArray();
12      indices = new DomainPointer[domainRanks.length];
13      indices[indices.length-1] = new DomainPointer();
14      for( int i = domainRanks.length-1; i > 0; i-- ) {
15        indices[i-1] = new DomainPointer();
16        indices[i-1].toRight = domainRanks[i] + indices[i].toRight;
17      }
18    }

20    @Override
21    public void execute() {
22      
23      // signal that execution is starting.
24      handler.start();
25      
26      try {
27      
28      // initialize the state and notify the handler.
29      for(int i = 0, used = 0; i < indices.length && used < rank; used += domainRanks[i++]) {
30        indices[i].index = used;
31        indices[i].count = Math.min(rank-used, domainRanks[i]);
32        for( int j = 0; j < indices[i].count; j++ ) {
33          last[indices[i].index+j] = domainValues[i][j];
34        }
35      }
36      handler.init(last);
37      handler.evaluate();
38      
39      // we are now at index 1, start iterating over the combinations.
40      for (long index = 1; index < size; index++) {
41  
42        // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION:
43        // (3,2,0,1,1)
44        // Position | 0 | 1 | 2 | 3 | 4
45        // Index | 0 | 3 | 5 | 5 | 6
46        // Count | 3 | 2 | 0 | 1 | 1
47        // ToRight | 9 | 6 | 5 | 2 | 0
48  
49        // advance the indices.
50        int cur = domainRanks.length - 1;
51        int remaining = 0;
52  
53        // move cur backwards to find the next item to update.
54        for (; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining + 1); remaining += indices[cur--].count);
55  
56        // decrement the items at cur.
57        indices[cur].count--;
58        remaining++;
59  
60        // move forward and update indices and next.
61        for (cur++; cur < indices.length; cur++) {
62          indices[cur].count = Math.min(remaining, domainRanks[cur]);
63          indices[cur].index = indices[cur - 1].index + indices[cur - 1].count;
64          remaining -= indices[cur].count;
65          for (int i = 0; i < indices[cur].count; i++) {
66            int replaceIndex = indices[cur].index + i;
67            T oldValue = last[replaceIndex];
68            last[indices[cur].index + i] = domainValues[cur][i];
69            
70            // fire the replace event.  This could be much faster with swaps.
71            handler.replace(last[replaceIndex], oldValue, replaceIndex);
72          }
73        }
74        handler.evaluate();
75      }
76      }
77      finally {
78        handler.end();
79      }
80    }
