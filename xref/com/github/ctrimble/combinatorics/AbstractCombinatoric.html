16  package com.github.ctrimble.combinatorics;
17  
18  import java.lang.reflect.Array;
19  import java.util.AbstractList;
20  
21  /**
22   * An abstract base class for combinatoric implementations.
23   * 
24   * @author Christian Trimble
25   *
26   * @param <T> the type of the elements being combined or permuted.
27   */
28  public abstract class AbstractCombinatoric<T> extends AbstractList<T[]>
29      implements Combinatoric<T> {
30  
31    protected int rank;
32    protected Multiset<T> domain;
33    protected T[][] domainValues;
34    protected long size;
35    protected CombMathUtils mathUtils;
36    protected Class<T> componentType;
37  
38    protected AbstractCombinatoric(int rank, T[] domain, CombMathUtils mathUtils) {
39      this.rank = rank;
40      this.domain = new FastMultiset<T>(rank, domain);
41      this.mathUtils = mathUtils;
42      this.size = computeSize(this.rank, this.domain);
43      this.componentType = getComponentType(domain);
44      this.domainValues = this.domain.toValueArray();
45    }
46    
47    @SuppressWarnings("unchecked")
48    protected static final <T> Class<T> getComponentType(T[] array) {
49      return (Class<T>)array.getClass().getComponentType();
50    }
51    
52    /**
53     * Returns a new array of type T with the length provided.
54     * 
55     * @return a new array of type T with the length provided.
56     */
57    @SuppressWarnings("unchecked")
58    protected final T[] newComponentArray(int length)
59    {
60      return (T[])Array.newInstance(componentType, length);
61    }
62  
63    @Override
64    public int getRank() {
65      return rank;
66    }
67  
68    @Override
69    public Multiset<T> getDomain() {
70      return domain;
71    }
72  
73    @Override
74    public long longSize() {
75      return size;
76    }
77  
78    @Override
79    public T[] get(int arg0) {
80      if (arg0 == 0) {
81        return iterator().next();
82      } else {
83        return subList(arg0, size()).iterator().next();
84      }
85    }
86  
87    @Override
88    public int size() {
89      if (size >= Integer.MAX_VALUE) {
90        return Integer.MAX_VALUE;
91      }
92      return (int) size;
93    }
94  
95    public abstract CombinatoricIterator<T> iterator();
96    protected abstract long computeSize(int rank, Multiset<T> domain);
97  
98    protected abstract class AbstractCombinatoricIterator
99        implements CombinatoricIterator<T>
100   {
101     protected long nextIndex = 0;
102     
103     protected AbstractCombinatoricIterator(long nextIndex)
104     {
105       this.nextIndex = nextIndex;
106     }
107     
108     @Override
109     public void add(T[] e) {
110       throw new UnsupportedOperationException();
111     }
112     
113     @Override
114     public long nextLongIndex() {
115       return nextIndex;
116     }
117 
118     @Override
119     public long previousLongIndex() {
120       return nextIndex-1;
121     }
122 
123     @Override
124     public int nextIndex() {
125       long nextLongIndex = nextLongIndex();
126       return nextLongIndex > Integer.MAX_VALUE
127           ? Integer.MAX_VALUE
128           : (int) nextLongIndex;
129     }
130 
131     @Override
132     public int previousIndex() {
133       long previousLongIndex = previousLongIndex();
134       return previousLongIndex > Integer.MAX_VALUE
135           ? Integer.MAX_VALUE
136           : (int) previousLongIndex;
137     }
138     
139     @Override
140     public boolean hasNext()
141     {
142       return nextIndex < size;
143     }
144     
145     @Override
146     public boolean hasPrevious()
147     {
148       return nextIndex > 0;
149     }
150 
151     @Override
152     public void remove() {
153       throw new UnsupportedOperationException();
154     }
155 
156     @Override
157     public void set(T[] e) {
158       throw new UnsupportedOperationException();
159     }
160   }
