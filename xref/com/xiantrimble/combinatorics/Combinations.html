21  /**
22   * A collection class that contains the combinations of a set of elements.  This implementation accounts for duplicate elements in the domain.
23   * 
24   * @author Christian Trimble
25   *
26   * @param <T> The type of element being combined.
27   */
28  public class Combinations<T>
29    extends AbstractCombinatoric<T>
30  {
31    /**
32     * Creates a new Combinations object over the domain.
33     * 
34     * @param k the length of the combinations.
35     * @param domain the elements that make up the combinations.
36     * @param mathUtil an instance of the math utilities.
37     */
38    public Combinations(int k, T[] domain, CombMathUtils mathUtil) {
39      super(k, domain, mathUtil);
40    }

42    /**
43     * Returns an iterator over this collection.
44     */
45    @Override
46    public CombinatoricIterator<T> iterator() {
47      return new CombinationIterator(0);
48    }

50    /**
51     * Computes the number of combinations for the specified length and domain.
52     * 
53     * @param k the length of the combinations.
54     * @param domain the elements to be combined.
55     */
56    @Override
57    protected long computeSize(int k, GroupedDomain<T> domain) {
58      return mathUtils.c(k, domain.toMultiplicity());
59    }
60    
61    @Override
62    public long longIndexOf(T[] element) {
63      // build a multiplicity array for the input, with zeros for elements not present.
64      int[] elementMultiplicity = new int[domainValues.length];
65      for( int i = 0, elementIndex = 0; i < domainValues.length; i++ ) {
66        elementMultiplicity[i] = 0;
67        for(;elementIndex < element.length && element[elementIndex].equals(domainValues[i][0]);elementIndex++) {
68          elementMultiplicity[i]++;
69        }
70      }
71      
72      // use the multiplicities to abstractly roll the index up.
73      // for each element type, compute the number of combinations that are not in the input elements.
74      int[] domainMultiplicity = this.domainMultiplicity.clone();
75      long index = 0;
76      int remaining = k;
77      for( int i = 0; i < domainValues.length-1 && remaining > 0; i++ ) {
78        // for empty elements.
79        int emptyElementStart = i;
80        int passedElements = 0;
81        while( elementMultiplicity[i] == 0 ) {
82          i++;
83          passedElements += domainMultiplicity[i];
84        }
85        if( emptyElementStart != i ) {
86          for( int j = 1; j <= Math.min(remaining, passedElements); j++ ) {
87            index += mathUtils.c(j, Arrays.copyOfRange(domainMultiplicity, emptyElementStart, i)) * mathUtils.c(remaining-j, Arrays.copyOfRange(domainMultiplicity, i, domainMultiplicity.length));
88          }
89        }
90        for( int w = emptyElementStart; w <= i; w++) {
91          domainMultiplicity[w] = 0;
92        }
93        // domainMultiplicity[i] = 0;
94        for( int j = domainValues[i].length; j > 0 && elementMultiplicity[i] < j; j-- ) {
95          index += mathUtils.c(remaining-j, domainMultiplicity);
96        }
97        remaining -= elementMultiplicity[i];
98      }
99      
100     return index;
101   }

103   /**
104    * The iterator that produces the individual elements for this Combinations object.
105    * 
106    * @author Christian Trimble
107    *
108    */
109   protected class CombinationIterator
110     extends AbstractCombinatoric<T>.AbstractCombinatoricIterator
111   {
112     protected T[] next;
113     protected T[] previous;
114     protected int[] domainMultiplicity;
115     protected DomainPointer[] indices;
116     
117     protected CombinationIterator(long nextIndex) {
118       super(nextIndex);
119       next = newComponentArray(k);
120       previous = newComponentArray(k);
121       domainMultiplicity = domain.toMultiplicity();
122       indices = new DomainPointer[domainMultiplicity.length];
123       indices[indices.length-1] = new DomainPointer();
124       for( int i = domainMultiplicity.length-1; i > 0; i-- ) {
125         indices[i-1] = new DomainPointer();
126         indices[i-1].toRight = domainMultiplicity[i] + indices[i].toRight;
127       }      
128     }
129 
130     /**
131      * Returns the next combination of the elements.
132      * 
133      * @throws NoSuchElementException if the end of the elements has been reached.
134      */
135     @Override
136     public T[] next() {
137       if( nextIndex >= size ) throw new NoSuchElementException(); // we may just want to do this in the next method.
138       
139       // reset the next array if needed.
140       if( nextIndex == 0 ) {
141         for(int i = 0, used = 0; i < indices.length && used < k; used += domainMultiplicity[i++]) {
142           indices[i].index = used;
143           indices[i].count = Math.min(k-used, domainMultiplicity[i]);
144           for( int j = 0; j < indices[i].count; j++ ) {
145             next[indices[i].index+j] = domainValues[i][j];
146           }
147         }
148       }
149       
150       // set both values to the the next value.
151       for( int i = 0; i < next.length; i++ ) previous[i] = next[i];
152       nextIndex++;
153       
154       if( nextIndex != size ) {
155       
156       // DIAGRAM OF INDICIES ARRAY: DOMAIN MULTIPLICITY: (3,3,1,3,2), CURRENT COMBINATION MULTIPLICITY: (3,2,0,1,1)
157       // Position  | 0 | 1 | 2 | 3 | 4
158       // Index     | 0 | 3 | 5 | 5 | 6
159       // Count     | 3 | 2 | 0 | 1 | 1
160       // ToRight   | 9 | 6 | 5 | 2 | 0
161 
162       // advance the indices.
163       int cur = domainMultiplicity.length - 1;
164       int remaining = 0;
165       
166       // move cur backwards to find the next item to update.
167       for( ; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining+1); remaining += indices[cur--].count );
168       
169       // decrement the items at cur.
170       indices[cur].count--;
171       remaining++;
172       
173       // move forward and update indices and next.
174       for(cur++; cur < indices.length; cur++) {
175         indices[cur].count = Math.min(remaining, domainMultiplicity[cur]);
176         indices[cur].index = indices[cur-1].index+indices[cur-1].count;
177         remaining -= indices[cur].count;
178         for( int i = 0; i < indices[cur].count; i++ ) {
179           next[indices[cur].index+i] = domainValues[cur][i];
180         }
181       }
182       }
183       // return previous, since we advanced past the next position.
184       return Arrays.copyOf(previous, previous.length);
185     }
186 
187     /**
188      * Returns the previous result.
189      * 
190      * @throws NoSuchElementException if there is no previous result to return.
191      */
192     @Override
193     public T[] previous() {
194       if( nextIndex <= 0 ) throw new NoSuchElementException(); // we may just want to do this in the next method.
195       
196       // set both values to the the previous value.
197       for( int i = 0; i < previous.length; i++ ) next[i] = previous[i];
198       nextIndex--;
199       
200       if( nextIndex > 0 ) {
201       
202       // DIAGRAM OF INDICIES ARRAY: DOMAIN MULTIPLICITY: (3,3,1,3,2), CURRENT COMBINATION MULTIPLICITY: (3,2,0,1,1)
203       // Position  | 0 | 1 | 2 | 3 | 4
204       // Index     | 0 | 3 | 5 | 5 | 6
205       // Count     | 3 | 2 | 0 | 1 | 1
206       // ToRight   | 9 | 6 | 5 | 2 | 0
207 
208       // advance the indices.
209       int cur = domainMultiplicity.length - 1;
210       int remaining = 0;
211 
212       // move cur backwards to find an item to increment.
213       for( ; cur > 0 && (remaining == 0 || indices[cur].count == domainMultiplicity[cur]); remaining += indices[cur--].count );
214 
215       // increment the items at cur.
216       indices[cur].count++;
217       previous[indices[cur].index+indices[cur].count-1] = domainValues[cur][indices[cur].count-1];
218       remaining--;
219        
220       for(cur++; cur < indices.length; cur++) {
221         indices[cur].count = Math.max(remaining-indices[cur].toRight, 0);
222         indices[cur].index = indices[cur-1].index+indices[cur-1].count;
223         remaining -= indices[cur].count;
224         for( int i = 0; i < indices[cur].count; i++ ) {
225           previous[indices[cur].index+i] = domainValues[cur][i];
226         }
227       }
228       }
229       
230       // return next, since we advanced past the previous position.
231       return Arrays.copyOf(next, next.length);
232     }
233   }

235   /**
236    * State information for each unique type in the domain.
237    * 
238    * @author Christian Trimble
239    */
240   private static class DomainPointer
241   {
242     /** The start index for this type in the current result. */
243     public int index = 0;
244     /** The number of elements of this type in the current result. */
245     public int count = 0;
246     /** The number of elements in the domain that are greater than this type. */
247     public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
248     /**
249      * Returns a string representation of this object.
250      * @return a string representation of this object.
251      */
252     public String toString()
253     {
254       return "{Index:"+index+",Count:"+count+",ToRight:"+toRight+"}";
255     }
256   }
