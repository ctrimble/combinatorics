16  package com.xiantrimble.combinatorics;
17  
18  /**
19   * An event handler for building a combination or permutation based on changes from the last combination or permutation.
20   * 
21   * Events are fired in this order:
22   * start()
23   * init()
24   * evaluate()
25   * (( init() | swap() | replace() ) evaluate() ) *
26   * end()
27   * 
28   * @author Christian Trimble
29   *
30   * @param <T>
31   */
32  public interface CombinatoricHandler<T> {
33    /**
34     * Called when events are going to start.
35     */
36    public void start();
37    
38    /**
39     * Signals the replacement of all state information.  This method will be called at least once after start(), but
40     * can also be called during iteration.
41     * 
42     * @param state the new combinatoric state.
43     */
44    public void init(T[] state);
45    
46    /**
47     * Signals the swapping of two elements in the current state.
48     * 
49     * @param newA the new value at position ai.  This value is at bi before this call.
50     * @param ai the new index of newA.
51     * @param newB the new value at position bi.  This value is at ai before this call.
52     * @param bi the new index of newB.
53     */
54    public void swap(T newA, int ai, T newB, int bi);
55    
56    /**
57     * Signals the replacement of a value in the current state.
58     * 
59     * @param newValue the new value at position i.
60     * @param oldValue the old value at position i.
61     * @param i the index that is being replaced.
62     */
63    public void replace(T newValue, T oldValue, int i);
64    
65    /**
66     * Signals that a new state has been reached and it is time to evaluate the combination or permutation.
67     */
68    public void evaluate();
69    
70    /**
71     * Called when events have finished.
72     */
73    public void end();
74  }
