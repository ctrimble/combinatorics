18  import java.util.Arrays;
19  import java.util.NoSuchElementException;

21  /**
22   * A CombinatoricEngine implementation for Permutations.
23   * 
24   * @author Christian Trimble
25   *
26   * @param <T> The type of element being permuted.
27   */
28  public class IndexBasedPermutationsEngine<T> extends AbstractCombinatoricEngine<T> {
29    protected TypePermutationState[] state;
30    protected int[] domainRanks;
31    protected T[] last;
32  
33    protected IndexBasedPermutationsEngine(int k, T[] domain, CombMathUtils mathUtils) {
34      super(k, domain, mathUtils);
35    }
36  
37    /**
38     * Computes the number of permutations for the specified length and domain.
39     * 
40     * @param k the length of the permutations.
41     * @param domain the elements to be permuted.
42     */
43    @Override
44    protected long computeSize(int k, GroupedDomain<T> domain) {
45      return mathUtils.p(k, domain.toMultiplicity());
46    }
47  
48    @Override
49    public void execute() {
50      handler.start();
51  
52      try {
53        int pastCombSize = 0;
54        last = Utils.newArray(componentType, k);
55        state = new TypePermutationState[domainMultiplicity.length];
56        int ni = 0; // index into the next solution array.
57        for( int dri = 0; dri < domainMultiplicity.length; dri++) {
58          state[dri] = new TypePermutationState();
59          state[dri].count = Math.min(domainMultiplicity[dri], k-ni);
60          for( int j = 0; j < state[dri].count; j++ ) {
61            last[ni++] = domain.get(dri).get(j);
62          }
63        }
64        for( int i = state.length - 2; i >= 0; i-- ) {
65          state[i].toRight = domainMultiplicity[i+1] + state[i+1].toRight;
66          state[i].activeToRight = state[i+1].activeToRight + state[i+1].count;
67          state[i].perms = mathUtils.p(state[i].count+state[i].activeToRight, new int[]{ state[i].count, state[i].activeToRight});
68        }
69  
70        handler.init(last);
71        handler.evaluate();
72        INDEX: for( int nextIndex = 1; nextIndex < size; nextIndex++ ) {
73        long index = nextIndex-pastCombSize;
74        int windowStart = 0;
75        int windowEnd = last.length;
76        TYPE: for( int i = 0; i < state.length-1; i++) {
77          if( state[i].count == 0 ) continue TYPE;
78          long typeCycles = index / (state[i].perms);
79          long typeIndex = index % (state[i].perms);
80          if( typeIndex == 0 ) {
81            index = typeCycles;
82            windowStart = typeCycles % 2 == 1 ? windowStart : windowStart + state[i].count;
83            windowEnd = typeCycles % 2 == 0 ? windowEnd : windowEnd - state[i].count;
84            continue TYPE;
85          }
86          
87          // we are now on the proper state for this index.  On even cycles we are moving down.
88          boolean down = typeCycles % 2 == 0;
89          typeIndex = down ? typeIndex : (state[i].perms) - typeIndex;
90          boolean elementDown = true;
91          int aCount = state[i].count;
92          int bCount = state[i].activeToRight;
93          int aIndex = 0;
94          ELEMENT: for( int elementIndex = 0; elementIndex < windowEnd - windowStart; elementIndex++) {
95            long perms = elementDown ? mathUtils.p((aCount-1)+bCount, new int[]{ aCount-1, bCount}) : mathUtils.p(aCount+bCount-1, new int[]{ aCount, bCount-1});
96            if( typeIndex == perms ) {
97              boolean moveMultiple = ((elementIndex % 2) != (aIndex % 2)) ^ !elementDown;
98              // this is the element, find the source and target.
99              int source = windowStart + elementIndex;
100             int target = windowStart + (moveMultiple ? elementIndex + aCount : elementIndex+1 );
101             T temp = last[source];
102             last[source] = last[target];
103             last[target] = temp;
104             handler.swap(last[target], target, last[source], source);
105             handler.evaluate();
106             continue INDEX;
107           }
108           else if( typeIndex < perms ^ !elementDown ) {
109             typeIndex = elementDown ? typeIndex : typeIndex - perms;
110             elementDown = (aIndex % 2) == (elementIndex % 2);
111             aCount--;
112             aIndex++;
113             continue ELEMENT;
114           }
115           else {
116             typeIndex = elementDown ? typeIndex - perms : typeIndex;
117             bCount--;
118             continue ELEMENT;
119           }
120         }
121       }
122       
123       pastCombSize = nextIndex;
124       // end of permutations for this combination reached.  Advance to the next combination.
125       int cur = state.length - 1;
126       int remaining = 0;
127       
128       // move cur backwards to find the next item to update.
129       for( ; cur > 0 && (state[cur].count == 0 || state[cur].toRight < remaining+1); remaining += state[cur--].count ); // back cur up to a position to increment.
130       
131       // decrement the items at cur.
132       state[cur].count--;
133       remaining++;
134       
135       // move forward and update all of the counts.
136       for(cur++; cur < state.length; cur++) {
137         state[cur].count = Math.min(remaining, domainMultiplicity[cur]);
138         remaining -= state[cur].count;
139       }
140       
141       for( int ri = 0, ni2 = 0;ri < domainMultiplicity.length; ri++) {
142         for( int j = 0; j < state[ri].count; j++ ) {
143           last[ni2++] = domain.get(ri).get(j);
144         }
145       }
146       
147       for( int i = state.length - 2; i >= 0; i-- ) {
148         state[i].activeToRight = state[i+1].activeToRight + state[i+1].count;
149         state[i].perms = mathUtils.p(state[i].count+state[i].activeToRight, new int[]{ state[i].count, state[i].activeToRight});
150       }
151       
152       handler.init(last);
153       handler.evaluate();
154       }
155     } finally {
156       handler.end();
157     }
158   }

160   private static class TypePermutationState {
161     public long perms;
162     public int activeToRight;
163     public Direction direction = Direction.DOWN;
164     public EntryPermutationState[] entryState;
165     public int count = 0; // the number of entries being used.
166     public int toRight = 0; // the number of items that can occur to the right.
167                             // Would be better with the rank array.
168     public String toString() {
169       return "{direction:" + direction + ", entries:" + Arrays.toString(entryState) + "}";
170     }
171   }
