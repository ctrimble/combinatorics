18  /**
19   * A CombinatoricEngine implementation for combinations.
20   * 
21   * @author Chritian Trimble
22   *
23   * @param <T> the type of the elements being combined or permuted.
24   */
25  public class CombinationsEngine<T> extends AbstractCombinatoricEngine<T> {
26    protected T[] last;
27    protected int[] domainMiltiplicity;
28    protected DomainPointer[] indices;
29    
30    protected CombinationsEngine(int k, T[] domain, CombMathUtils mathUtils) {
31      super(k, domain, mathUtils);
32      last = Utils.newArray(componentType, k);
33      domainMiltiplicity = this.domain.toMultiplicity();
34      indices = new DomainPointer[domainMiltiplicity.length];
35      indices[indices.length-1] = new DomainPointer();
36      for( int i = domainMiltiplicity.length-1; i > 0; i-- ) {
37        indices[i-1] = new DomainPointer();
38        indices[i-1].toRight = domainMiltiplicity[i] + indices[i].toRight;
39      }
40    }
41  
42    @Override
43    public void execute() {
44      
45      // signal that execution is starting.
46      handler.start();
47      
48      try {
49      
50      // initialize the state and notify the handler.
51      for(int i = 0, used = 0; i < indices.length && used < k; used += domainMiltiplicity[i++]) {
52        indices[i].index = used;
53        indices[i].count = Math.min(k-used, domainMiltiplicity[i]);
54        for( int j = 0; j < indices[i].count; j++ ) {
55          last[indices[i].index+j] = domainValues[i][j];
56        }
57      }
58      handler.init(last);
59      handler.evaluate();
60      
61      // we are now at index 1, start iterating over the combinations.
62      for (long index = 1; index < size; index++) {
63  
64        // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION:
65        // (3,2,0,1,1)
66        // Position | 0 | 1 | 2 | 3 | 4
67        // Index | 0 | 3 | 5 | 5 | 6
68        // Count | 3 | 2 | 0 | 1 | 1
69        // ToRight | 9 | 6 | 5 | 2 | 0
70  
71        // advance the indices.
72        int cur = domainMiltiplicity.length - 1;
73        int remaining = 0;
74  
75        // move cur backwards to find the next item to update.
76        for (; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining + 1); remaining += indices[cur--].count);
77  
78        // decrement the items at cur.
79        indices[cur].count--;
80        remaining++;
81  
82        // move forward and update indices and next.
83        for (cur++; cur < indices.length; cur++) {
84          indices[cur].count = Math.min(remaining, domainMiltiplicity[cur]);
85          indices[cur].index = indices[cur - 1].index + indices[cur - 1].count;
86          remaining -= indices[cur].count;
87          for (int i = 0; i < indices[cur].count; i++) {
88            int replaceIndex = indices[cur].index + i;
89            T oldValue = last[replaceIndex];
90            last[indices[cur].index + i] = domainValues[cur][i];
91            
92            // fire the replace event.  This could be much faster with swaps.
93            handler.replace(last[replaceIndex], oldValue, replaceIndex);
94          }
95        }
96        handler.evaluate();
97      }
98      }
99      finally {
100       handler.end();
101     }
102   }

104   /**
105    * Computes the number of combinations for the specified length and domain.
106    * 
107    * @param k the length of the combination.
108    * @param domain the elements to be combined.
109    */
110   @Override
111   protected long computeSize(int k, GroupedDomain<T> domain) {
112     return mathUtils.c(k, domain.toMultiplicity());
113   }

115   /**
116    * State information for each unique type in the domain.
117    * 
118    * @author Christian Trimble
119    */
120   private static class DomainPointer
121   {
122     /** The start index for this type in the current result. */
123     public int index = 0;
124     /** The number of elements of this type in the current result. */
125     public int count = 0;
126     /** The number of elements in the domain that are greater than this type. */
127     public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
128     /**
129      * Returns a string representation of this object.
130      * @return a string representation of this object.
131      */
132     public String toString()
133     {
134       return "{Index:"+index+",Count:"+count+",ToRight:"+toRight+"}";
135     }
136   }
