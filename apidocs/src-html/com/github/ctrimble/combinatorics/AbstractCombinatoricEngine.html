001    package com.github.ctrimble.combinatorics;
002    
003    public abstract class AbstractCombinatoricEngine<T>
004        implements CombinatoricEngine<T>
005    {
006      protected CombinatoricHandler<T> handler;
007      protected int rank;
008      protected Multiset<T> domain;
009      protected T[][] domainValues;
010      protected long size;
011      protected CombMathUtils mathUtils;
012      protected Class<T> componentType;
013    
014      protected AbstractCombinatoricEngine(int rank, T[] domain, CombMathUtils mathUtils) {
015        this.rank = rank;
016        this.domain = new FastMultiset<T>(rank, domain);
017        this.mathUtils = mathUtils;
018        this.size = computeSize(this.rank, this.domain);
019        this.componentType = Utils.getComponentType(domain);
020        this.domainValues = this.domain.toValueArray();
021      }
022    
023      @Override
024      public void setHandler(CombinatoricHandler<T> handler) {
025        this.handler = handler;
026      }
027    
028    
029      @Override
030      public int size() {
031        if (size >= Integer.MAX_VALUE) {
032          return Integer.MAX_VALUE;
033        }
034        return (int) size;
035      }
036    
037      protected abstract long computeSize(int rank, Multiset<T> domain);
038    }
