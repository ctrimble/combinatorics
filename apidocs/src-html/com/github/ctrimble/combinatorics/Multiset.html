016    package com.github.ctrimble.combinatorics;
017    
018    import java.util.List;
019    
020    /**
021     * A representation of a set of elements where the like elements have been grouped.  Usually, a 
022     * multiset would just contain information about the quantities of the distinct types in a collection,
023     * but this implementation also holds the actual elements.
024     * 
025     * @author Christian Trimble
026     *
027     * @param <E> the element type of the collection or array that this multiset represents.
028     */
029    public interface Multiset<E> extends List<List<E>>
030    {
031      /**
032       * Returns the total number of elements in this multiset.
033       * 
034       * @return the total number of elements in this multiset.
035       */
036      public int getRank();
037      
038      /**
039       * Returns an array of the element ranks for this multiset.
040       * 
041       * @return an array of the element ranks for this multiset.
042       */
043      public int[] toRankArray();
044      
045      /**
046       * Returns a two dimensional array representation of this multiset.
047       * 
048       * @return a two dimensional array representation of this multiset.
049       */
050      public E[][] toValueArray();
051    }
