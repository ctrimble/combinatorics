001    package com.github.ctrimble.combinatorics;
002    
003    import java.util.Arrays;
004    
005    public class PermutationsEngine<T> extends AbstractCombinatoricEngine<T> {
006      protected TypePermutationState[] state;
007      protected int[] domainRanks;
008      protected T[] last;
009    
010      protected PermutationsEngine(int rank, T[] domain, CombMathUtils mathUtils) {
011        super(rank, domain, mathUtils);
012      }
013    
014      /**
015       * Computes the number of permutations for the specified rank and domain.
016       * 
017       * @param rank
018       *          the rank of the permutation.
019       * @param domain
020       *          the multiset containing the elements to be permuted.
021       */
022      @Override
023      protected long computeSize(int rank, Multiset<T> domain) {
024        return mathUtils.p(rank, domain.toRankArray());
025      }
026    
027      @Override
028      public void execute() {
029        handler.start();
030    
031        try {
032    
033          last = Utils.newArray(componentType, rank);
034          domainRanks = domain.toRankArray();
035          state = new TypePermutationState[domainRanks.length];
036          int ni = 0;
037          for (int ri = 0; ri < domainRanks.length; ri++) {
038            state[ri] = new TypePermutationState();
039            state[ri].count = Math.min(domainRanks[ri], rank - ni);
040            state[ri].entryState = new EntryPermutationState[domainRanks[ri]];
041            for (int j = 0; j < state[ri].entryState.length; j++) {
042              state[ri].entryState[j] = new EntryPermutationState(j);
043              if (j < state[ri].count) {
044                last[ni++] = domain.get(ri).get(j);
045              }
046            }
047          }
048          for (int i = state.length - 2; i >= 0; i--) {
049            state[i].toRight = state[i].entryState.length + state[i + 1].toRight;
050          }
051    
052          handler.init(last);
053          handler.evaluate();
054    
055          ITERATE : for (long index = 1; index < size; index++) {
056            int windowStart = 0;
057            int windowEnd = last.length;
058            int swapSource = 0;
059            int swapTarget = 0;
060            TYPE : for (int i = 0; i < state.length; i++) {
061              switch (state[i].direction) {
062                case DOWN :
063                  // scan the entries from back to front, looking for the first item
064                  // to move.
065                  ENTRY : for (int j = state[i].count - 1; j >= 0; j--) {
066                    switch (state[i].entryState[j].direction) {
067                      case DOWN :
068                        if (state[i].entryState[j].index < (windowEnd - 1) - ((state[i].count - 1) - j)) {
069                          // track the source index for the swap.
070                          swapSource = windowStart + state[i].entryState[j].index;
071    
072                          // increment the index and deal with moving over like
073                          // elements
074                          state[i].entryState[j].index++;
075                          for (; j < state[i].count - 1 && state[i].entryState[j].index == state[i].entryState[j + 1].index; j++) {
076                            state[i].entryState[j + 1].index++;
077                            state[i].entryState[j + 1].direction = Direction.DOWN;
078                          }
079    
080                          // track the target index for the swap.
081                          swapTarget = windowStart + state[i].entryState[j].index;
082                          break TYPE;
083                        }
084                        state[i].entryState[j].direction = Direction.UP;
085                        break;
086                      case UP :
087                        int startIndex = j;
088                        for (; state[i].entryState[j].index == state[i].entryState[j - 1].index + 1; j--) {
089                          // if this entry is moving down, then switch case
090                          // statements.
091                          if (state[i].entryState[j - 1].direction == Direction.DOWN) {
092                            continue ENTRY;
093                          }
094                        }
095                        swapSource = windowStart + state[i].entryState[startIndex].index;
096                        state[i].entryState[j].direction = Direction.UP;
097                        state[i].entryState[j].index--;
098                        swapTarget = windowStart + state[i].entryState[j].index;
099                        for (j++; j <= startIndex; j++) {
100                          state[i].entryState[j].direction = Direction.DOWN;
101                          state[i].entryState[j].index--;
102                        }
103                        break TYPE;
104                    }
105                  }
106                  // none of the entries can move down, switch directions.
107                  state[i].direction = Direction.UP;
108                  windowEnd -= state[i].count;
109                  if (i < state.length - 1)
110                    continue TYPE;
111                  break;
112                case UP :
113                  // scan the entries from front to back, looking for the first item
114                  // to move.
115                  ENTRY : for (int j = 0; j < state[i].count; j++) {
116                    switch (state[i].entryState[j].direction) {
117                      case UP :
118                        if (state[i].entryState[j].index > j) {
119                          // track the source index of the swap.
120                          swapSource = state[i].entryState[j].index + windowStart;
121    
122                          // decrement the index and deal with moving over like
123                          // elements.
124                          state[i].entryState[j].index--;
125                          for (; j > 0 && state[i].entryState[j].index == state[i].entryState[j - 1].index; j--) {
126                            state[i].entryState[j - 1].index--;
127                            state[i].entryState[j - 1].direction = Direction.UP;
128                          }
129    
130                          // track the target index for the swap.
131                          swapTarget = state[i].entryState[j].index + windowStart;
132                          break TYPE;
133                        }
134                        state[i].entryState[j].direction = Direction.DOWN;
135                        break;
136                      case DOWN :
137                        int startIndex = j;
138                        for (; state[i].entryState[j].index == state[i].entryState[j + 1].index - 1; j++) {
139                          // if this entry is moving down, then switch case
140                          // statements.
141                          if (state[i].entryState[j + 1].direction == Direction.UP) {
142                            continue ENTRY;
143                          }
144                        }
145                        swapSource = windowStart + state[i].entryState[startIndex].index;
146                        state[i].entryState[j].direction = Direction.DOWN;
147                        state[i].entryState[j].index++;
148                        swapTarget = windowStart + state[i].entryState[j].index;
149                        for (j--; j >= startIndex; j--) {
150                          state[i].entryState[j].direction = Direction.UP;
151                          state[i].entryState[j].index++;
152                        }
153                        break TYPE;
154                    }
155                  }
156                  state[i].direction = Direction.DOWN;
157                  windowStart += state[i].count;
158                  if (i < state.length - 1)
159                    continue TYPE;
160                  break;
161              }
162    
163              // end of permutations for this combination reached. Advance to the
164              // next
165              // combination.
166              int cur = state.length - 1;
167              int remaining = 0;
168    
169              // move cur backwards to find the next item to update.
170              for (; cur > 0 && (state[cur].count == 0 || state[cur].toRight < remaining + 1); remaining += state[cur--].count);
171    
172              // decrement the items at cur.
173              state[cur].count--;
174              remaining++;
175    
176              // move forward and update all of the counts.
177              for (cur++; cur < state.length; cur++) {
178                state[cur].count = Math.min(remaining, domainRanks[cur]);
179                remaining -= state[cur].count;
180              }
181    
182              // for now, totally reset next. Making this an incremental update will
183              // help when the rank is much smaller than the number of entries.
184              ni = 0;
185              for (int ri = 0; ri < domainRanks.length; ri++) {
186                state[ri].direction = Direction.DOWN;
187                for (int k = 0; k < state[ri].entryState.length; k++) {
188                  state[ri].entryState[k].index = k;
189                  state[ri].entryState[k].direction = Direction.DOWN;
190                  if (k < state[ri].count) {
191                    last[ni++] = domain.get(ri).get(k);
192                  }
193                }
194              }
195              handler.init(last);
196              handler.evaluate();
197              continue ITERATE;
198            }
199    
200            // swap the indices.
201            T target = last[swapTarget];
202            last[swapTarget] = last[swapSource];
203            last[swapSource] = target;
204    
205            handler.swap(last[swapTarget], swapTarget, last[swapSource], swapSource);
206            handler.evaluate();
207          }
208        } finally {
209          handler.end();
210        }
211      }
212    
213      private static class TypePermutationState {
214        public Direction direction = Direction.DOWN;
215        public EntryPermutationState[] entryState;
216        public int count = 0; // the number of entries being used.
217        public int toRight = 0; // the number of items that can occur to the right.
218                                // Would be better with the rank array.
219        public String toString() {
220          return "{direction:" + direction + ", entries:" + Arrays.toString(entryState) + "}";
221        }
222      }
223    
224      private static class EntryPermutationState {
225        public EntryPermutationState(int index) {
226          this.index = index;
227        }
228        public Direction direction = Direction.DOWN;
229        public int index = 0;
230        public String toString() {
231          return "{direction:" + direction + ", index:" + index + "}";
232        }
233      }
234    
235      private static enum Direction {
236        UP, DOWN
237      }
238    }
