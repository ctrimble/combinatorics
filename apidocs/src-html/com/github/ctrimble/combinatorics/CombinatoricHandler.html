001    package com.github.ctrimble.combinatorics;
002    
003    /**
004     * An event handler for building a combination or permutation based on changes from the last combination or permutation.
005     * 
006     * Events are fired in this order:
007     * start()
008     * init()
009     * evaluate()
010     * (( init() | swap() | replace() ) evaluate() ) *
011     * end()
012     * 
013     * @author Christian Trimble
014     *
015     * @param <T>
016     */
017    public interface CombinatoricHandler<T> {
018      /**
019       * Called when events are going to start.
020       */
021      public void start();
022      
023      /**
024       * Signals the replacement of all state information.  This method will be called at least once after start(), but
025       * can also be called during iteration.
026       * 
027       * @param state the new combinatoric state.
028       */
029      public void init(T[] state);
030      
031      /**
032       * Signals the swapping of two elements in the current state.
033       * 
034       * @param newA the new value at position ai.  This value is at bi before this call.
035       * @param ai the new index of newA.
036       * @param newB the new value at position bi.  This value is at ai before this call.
037       * @param bi the new index of newB.
038       */
039      public void swap(T newA, int ai, T newB, int bi);
040      
041      /**
042       * Signals the replacement of a value in the current state.
043       * 
044       * @param newValue the new value at position i.
045       * @param oldValue the old value at position i.
046       * @param i the index that is being replaced.
047       */
048      public void replace(T newValue, T oldValue, int i);
049      
050      /**
051       * Signals that a new state has been reached and it is time to evaluate the combination or permutation.
052       */
053      public void evaluate();
054      
055      /**
056       * Called when events have finished.
057       */
058      public void end();
059    }
