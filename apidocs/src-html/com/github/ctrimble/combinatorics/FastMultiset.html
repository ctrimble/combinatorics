016    package com.github.ctrimble.combinatorics;
017    
018    import java.util.Arrays;
019    import java.util.List;
020    
021    import javolution.util.FastList;
022    
023    /**
024     * An implementation of Multiset, based on Javolutions FastList.
025     * 
026     * @author Christian Trimble
027     *
028     * @param <E> the element type of the domain.
029     */
030    @SuppressWarnings("serial")
031    public class FastMultiset<E>
032      extends FastList<List<E>>
033      implements Multiset<E>
034    {
035      /** The total of all the element ranks of this multiset. */
036      protected int rank = 0;
037      /** The class for the elements in this multiset. */
038      protected Class<E> componentType;
039    
040      /**
041       * Creates a new FastMultiset for the specified domain.
042       * 
043       * @param domain the elements that make up this multiset.
044       */
045      public FastMultiset(E[] domain) {
046        this(Integer.MAX_VALUE, domain);
047      }
048      
049      /**
050       * Creates a new FastMultiSet for the specified domain.  If any type has more than maxTypeRank elements,
051       * then those elements will be excluded from the multiset.
052       * 
053       * @param maxTypeRank the maximum number of elements for any unique value in the domain.
054       * @param domain the elements that make up this multiset.
055       */
056      public FastMultiset(int maxTypeRank, E[] domain)
057      {
058        componentType = Utils.getComponentType(domain);
059        E[] newDomain = Arrays.copyOf(domain, domain.length);
060        Arrays.sort(newDomain);
061        rank = 0;
062        for( int i = 0; i < newDomain.length;) {
063          int cur = i;
064          FastList<E> elements = new FastList<E>();
065          for( ; i < newDomain.length && newDomain[cur].equals(newDomain[i]); i++) {
066            if( i - cur < maxTypeRank ) {
067              elements.addFirst(newDomain[i]);
068            }
069          }
070          add(elements.unmodifiable());
071          rank += elements.size();
072        }
073      }
074    
075      @Override
076      public int getRank() {
077        return rank;
078      }
079    
080      @Override
081      public int[] toRankArray() {
082        int[] rankArray = new int[size()];
083        int i = 0;
084        for(List<E> element: this) {
085          rankArray[i++] = element.size(); 
086        }
087        return rankArray;
088      }
089    
090      @Override
091      public E[][] toValueArray() {
092        E[][] valueArray = Utils.newArray(componentType, size(), 0);
093        for( int i = 0; i < size(); i++ ) {
094          valueArray[i] = get(i).toArray(Utils.newArray(componentType, get(i).size()));
095        }
096        return valueArray;
097      }
098    }
