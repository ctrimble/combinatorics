001    package com.github.ctrimble.combinatorics;
002    
003    public class CombinationsEngine<T> extends AbstractCombinatoricEngine<T> {
004      protected T[] last;
005      protected int[] domainRanks;
006      protected DomainPointer[] indices;
007      
008      protected CombinationsEngine(int rank, T[] domain, CombMathUtils mathUtils) {
009        super(rank, domain, mathUtils);
010        last = Utils.newArray(componentType, rank);
011        domainRanks = this.domain.toRankArray();
012        indices = new DomainPointer[domainRanks.length];
013        indices[indices.length-1] = new DomainPointer();
014        for( int i = domainRanks.length-1; i > 0; i-- ) {
015          indices[i-1] = new DomainPointer();
016          indices[i-1].toRight = domainRanks[i] + indices[i].toRight;
017        }
018      }
019    
020      @Override
021      public void execute() {
022        
023        // signal that execution is starting.
024        handler.start();
025        
026        try {
027        
028        // initialize the state and notify the handler.
029        for(int i = 0, used = 0; i < indices.length && used < rank; used += domainRanks[i++]) {
030          indices[i].index = used;
031          indices[i].count = Math.min(rank-used, domainRanks[i]);
032          for( int j = 0; j < indices[i].count; j++ ) {
033            last[indices[i].index+j] = domainValues[i][j];
034          }
035        }
036        handler.init(last);
037        handler.evaluate();
038        
039        // we are now at index 1, start iterating over the combinations.
040        for (long index = 1; index < size; index++) {
041    
042          // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION:
043          // (3,2,0,1,1)
044          // Position | 0 | 1 | 2 | 3 | 4
045          // Index | 0 | 3 | 5 | 5 | 6
046          // Count | 3 | 2 | 0 | 1 | 1
047          // ToRight | 9 | 6 | 5 | 2 | 0
048    
049          // advance the indices.
050          int cur = domainRanks.length - 1;
051          int remaining = 0;
052    
053          // move cur backwards to find the next item to update.
054          for (; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining + 1); remaining += indices[cur--].count);
055    
056          // decrement the items at cur.
057          indices[cur].count--;
058          remaining++;
059    
060          // move forward and update indices and next.
061          for (cur++; cur < indices.length; cur++) {
062            indices[cur].count = Math.min(remaining, domainRanks[cur]);
063            indices[cur].index = indices[cur - 1].index + indices[cur - 1].count;
064            remaining -= indices[cur].count;
065            for (int i = 0; i < indices[cur].count; i++) {
066              int replaceIndex = indices[cur].index + i;
067              T oldValue = last[replaceIndex];
068              last[indices[cur].index + i] = domainValues[cur][i];
069              
070              // fire the replace event.  This could be much faster with swaps.
071              handler.replace(last[replaceIndex], oldValue, replaceIndex);
072            }
073          }
074          handler.evaluate();
075        }
076        }
077        finally {
078          handler.end();
079        }
080      }
081    
082      /**
083       * Computes the number of combinations for the specified rank and domain.
084       * 
085       * @param rank the rank of the permutation.
086       * @param domain the multiset containing the elements to be permuted.
087       */
088      @Override
089      protected long computeSize(int rank, Multiset<T> domain) {
090        return mathUtils.c(rank, domain.toRankArray());
091      }
092    
093      /**
094       * State information for each unique type in the domain.
095       * 
096       * @author Christian Trimble
097       */
098      private static class DomainPointer
099      {
100        /** The start index for this type in the current result. */
101        public int index = 0;
102        /** The number of elements of this type in the current result. */
103        public int count = 0;
104        /** The number of elements in the domain that are greater than this type. */
105        public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
106        /**
107         * Returns a string representation of this object.
108         * @return a string representation of this object.
109         */
110        public String toString()
111        {
112          return "{Index:"+index+",Count:"+count+",ToRight:"+toRight+"}";
113        }
114      }
115    }
