016    package com.github.ctrimble.combinatorics;
017    
018    /**
019     * Math methods for dealing with complex combinatoric problems with duplicates involved.
020     * 
021     * @author Christian Trimble
022     *
023     */
024    public interface CombMathUtils {
025    
026      /**
027       * Returns the number of combinations of length k from a multiset.
028       * 
029       * @param k the rank of the combinations to count.
030       * @param m the multiset of the domain elements.
031       * @return the number of combinations of length k from the multiset m.
032       */
033      long c(int k, int... m);
034    
035      /**
036       * Counts the number of permutations of length k from a multiset.
037       * 
038       * @param k the rank of the permutations to count.
039       * @param m the multiset of the domain elements.
040       * @return the number of permutations of length k from the multiset m.
041       */
042      long p(int k, int... m);
043      
044    }
