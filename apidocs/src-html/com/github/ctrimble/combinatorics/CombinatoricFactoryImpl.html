001    package com.github.ctrimble.combinatorics;
002    
003    public class CombinatoricFactoryImpl
004      implements CombinatoricFactory
005    {
006      private CombMathUtils mathUtils = new CombMathUtilsImpl();
007      
008      @Override
009      public <T> Combinations<T> createCombinations(int rank, T... domain) {
010        return new Combinations<T>(rank, domain, getMathUtils());
011      }
012    
013      @Override
014      public <T> Permutations<T> createPermutations(int rank, T... domain) {
015        return new Permutations<T>(rank, domain, getMathUtils());
016      }
017    
018      @Override
019      public <T> Multiset<T> createMultiset(T... domain) {
020        return new FastMultiset<T>(domain);
021      }
022    
023      @Override
024      public <T> Multiset<T> createMultiset(int maxElementRank, T... domain) {
025        return new FastMultiset<T>(maxElementRank, domain);
026      }
027    
028      @Override
029      public CombMathUtils getMathUtils() {
030        return mathUtils;
031      }
032    
033      @Override
034      public <T> CombinationsEngine<T> createCombinationsEngine(int rank, T... domain) {
035        return new CombinationsEngine<T>(rank, domain, getMathUtils());
036      }
037    
038      @Override
039      public <T> PermutationsEngine<T> createPermutationsEngine(int rank, T... domain) {
040        return new PermutationsEngine<T>(rank, domain, getMathUtils());
041      }
042    
043    }
