016    package com.github.ctrimble.combinatorics;
017    
018    import java.util.Arrays;
019    import java.util.NoSuchElementException;
020    
021    /**
022     * A collection class that contains the combinations of a set of elements.  This implementation accounts for duplicate elements in the domain.
023     * 
024     * @author Christian Trimble
025     *
026     * @param <T> The type of element being combined.
027     */
028    public class Combinations<T>
029      extends AbstractCombinatoric<T>
030    {
031      /**
032       * Creates a new Combinations object over the domain.
033       * 
034       * @param rank the length of the combinations.
035       * @param domain the elements that make up the combinations.
036       * @param mathUtil an instance of the math utilities.
037       */
038      public Combinations(int rank, T[] domain, CombMathUtils mathUtil) {
039        super(rank, domain, mathUtil);
040      } 
041      
042      @Override
043      public CombinatoricIterator<T> iterator() {
044        return new CombinationIterator(0);
045      }
046    
047      /**
048       * Computes the number of combinations for the specified rank and domain.
049       * 
050       * @param rank the rank of the permutation.
051       * @param domain the multiset containing the elements to be permuted.
052       */
053      @Override
054      protected long computeSize(int rank, Multiset<T> domain) {
055        return mathUtils.c(rank, domain.toRankArray());
056      }
057      
058      /**
059       * The iterator that produces the individual elements for this Combinations object.
060       * 
061       * @author Christian Trimble
062       *
063       */
064      protected class CombinationIterator
065        extends AbstractCombinatoric<T>.AbstractCombinatoricIterator
066      {
067        protected T[] next;
068        protected T[] previous;
069        protected int[] domainRanks;
070        protected DomainPointer[] indices;
071        
072        protected CombinationIterator(long nextIndex) {
073          super(nextIndex);
074          next = newComponentArray(rank);
075          previous = newComponentArray(rank);
076          domainRanks = domain.toRankArray();
077          indices = new DomainPointer[domainRanks.length];
078          indices[indices.length-1] = new DomainPointer();
079          for( int i = domainRanks.length-1; i > 0; i-- ) {
080            indices[i-1] = new DomainPointer();
081            indices[i-1].toRight = domainRanks[i] + indices[i].toRight;
082          }      
083        }
084    
085        @Override
086        public T[] next() {
087          if( nextIndex >= size ) throw new NoSuchElementException(); // we may just want to do this in the next method.
088          
089          // reset the next array if needed.
090          if( nextIndex == 0 ) {
091            for(int i = 0, used = 0; i < indices.length && used < rank; used += domainRanks[i++]) {
092              indices[i].index = used;
093              indices[i].count = Math.min(rank-used, domainRanks[i]);
094              for( int j = 0; j < indices[i].count; j++ ) {
095                next[indices[i].index+j] = domainValues[i][j];
096              }
097            }
098          }
099          
100          // set both values to the the next value.
101          for( int i = 0; i < next.length; i++ ) previous[i] = next[i];
102          nextIndex++;
103          
104          if( nextIndex != size ) {
105          
106          // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION: (3,2,0,1,1)
107          // Position  | 0 | 1 | 2 | 3 | 4
108          // Index     | 0 | 3 | 5 | 5 | 6
109          // Count     | 3 | 2 | 0 | 1 | 1
110          // ToRight   | 9 | 6 | 5 | 2 | 0
111    
112          // advance the indices.
113          int cur = domainRanks.length - 1;
114          int remaining = 0;
115          
116          // move cur backwards to find the next item to update.
117          for( ; cur > 0 && (indices[cur].count == 0 || indices[cur].toRight < remaining+1); remaining += indices[cur--].count ); // back cur up to a position to increment.
118          
119          // decrement the items at cur.
120          indices[cur].count--;
121          remaining++;
122          
123          // move forward and update indices and next.
124          for(cur++; cur < indices.length; cur++) {
125            indices[cur].count = Math.min(remaining, domainRanks[cur]);
126            indices[cur].index = indices[cur-1].index+indices[cur-1].count;
127            remaining -= indices[cur].count;
128            for( int i = 0; i < indices[cur].count; i++ ) {
129              next[indices[cur].index+i] = domainValues[cur][i];
130            }
131          }
132          }
133          // return previous, since we advanced past the next position.
134          return Arrays.copyOf(previous, previous.length);
135        }
136    
137        @Override
138        public T[] previous() {
139          if( nextIndex <= 0 ) throw new NoSuchElementException(); // we may just want to do this in the next method.
140          
141          // set both values to the the previous value.
142          for( int i = 0; i < previous.length; i++ ) next[i] = previous[i];
143          nextIndex--;
144          
145          if( nextIndex > 0 ) {
146          
147          // DIAGRAM OF INDICIES ARRAY: MULTISET: (3,3,1,3,2), CURRENT COMBINATION: (3,2,0,1,1)
148          // Position  | 0 | 1 | 2 | 3 | 4
149          // Index     | 0 | 3 | 5 | 5 | 6
150          // Count     | 3 | 2 | 0 | 1 | 1
151          // ToRight   | 9 | 6 | 5 | 2 | 0
152    
153          // advance the indices.
154          int cur = domainRanks.length - 1;
155          int remaining = 0;
156    
157          // move cur backwards to find an item to increment.
158          for( ; cur > 0 && (remaining == 0 || indices[cur].count == domainRanks[cur]); remaining += indices[cur--].count );
159    
160          // decrement the items at cur.
161          indices[cur].count++;
162          previous[indices[cur].index+indices[cur].count-1] = domainValues[cur][indices[cur].count-1];
163          remaining--;
164           
165          for(cur++; cur < indices.length; cur++) {
166            indices[cur].count = Math.max(remaining-indices[cur].toRight, 0);
167            indices[cur].index = indices[cur-1].index+indices[cur-1].count;
168            remaining -= indices[cur].count;
169            for( int i = 0; i < indices[cur].count; i++ ) {
170              previous[indices[cur].index+i] = domainValues[cur][i];
171            }
172          }
173          }
174          
175          // return next, since we advanced past the previous position.
176          return Arrays.copyOf(next, next.length);
177        }
178      }
179      
180      /**
181       * State information for each unique type in the domain.
182       * 
183       * @author Christian Trimble
184       */
185      private static class DomainPointer
186      {
187        /** The start index for this type in the current result. */
188        public int index = 0;
189        /** The number of elements of this type in the current result. */
190        public int count = 0;
191        /** The number of elements in the domain that are greater than this type. */
192        public int toRight = 0; // the number of items that can occur to the right.  Would be better with the rank array.
193        /**
194         * Returns a string representation of this object.
195         * @return a string representation of this object.
196         */
197        public String toString()
198        {
199          return "{Index:"+index+",Count:"+count+",ToRight:"+toRight+"}";
200        }
201      }
202    }
